42 template < class iter_t,
43  typename compare = std::less <typename iterator_traits<iter_t>::value_type> >
44 struct parallel_sort_comp
45 { //------------------------- begin ----------------------
46  static const size_t Min_Parallel = 65536 ;
47  compare comp;
48 
49  //------------------------------------------------------------------------
50  // function : parallel_sort_comp
51  /// @brief constructor of the struct
52  /// @param [in] first : iterator to the first element to sort
53  /// @param [in] last : iterator to the next element after the last
54  /// @param [in] comp : object for to compare
55  /// @param [in] NT : variable indicating the number of threads used in the
56  /// sorting
57  //------------------------------------------------------------------------
58  parallel_sort_comp ( iter_t first, iter_t last, compare comp1= compare ())
59  :comp(comp1)
60  { //------------------------- begin ----------------------
61  auto N = last - first;
62  assert ( N >=0);
63 
64  if ( (size_t)N < Min_Parallel )
65  { intro_sort ( first, last,comp) ;
66  return ;
67  } ;
68  //------------------- check if sort ----------------------------------
69  bool SW = true ;
70  for ( iter_t it1 = first, it2 = first+1 ;
71  it2 != last and (SW = not comp(*it2,*it1));it1 = it2++);
72  if (SW) return ;
73 
74  //-------------------------------------------------------------------
75  hpx::future<void> Rt = hpx::async( &parallel_sort_comp::sort_thread,
76  this ,first , last );
77  Rt.get() ;
78 
79  };

81  //------------------------------------------------------------------------
82  // function : sort_thread
83  /// @brief this function is the work asigned to each thread in the parallel
84  /// process
85  /// @param [in] first : iterator to the first element to sort
86  /// @param [in] last : iterator to the next element after the last
87  /// @param [in] level : level of depth
88  //------------------------------------------------------------------------
89  hpx::future <void> sort_thread(iter_t first, iter_t last )
90  { //------------------------- begin ----------------------
91  size_t N = last - first ;
92  if ( N <= Min_Parallel)
93  { return
94  hpx::async([this,first,last]() { intro_sort(first, last, comp);});
95  };
96  //------------------- check if sort ------------------------------
97  bool SW = true ;
98  for ( iter_t it1 = first, it2 = first+1 ;
99  it2 != last and (SW = not comp(*it2,*it1));it1 = it2++);
100  if (SW) return hpx::make_ready_future();
101 
102  //----------------------------------------------------------------
103  // split
104  //----------------------------------------------------------------
105  typedef typename iterator_traits<iter_t>::value_type value_t ;
106 
107 
108  //---------------------- pivot select ----------------------------
109  iter_t itA = first +1 ;
110  iter_t itB = first + ( size_t (N ) >>1 ) ;
111  iter_t itC = last -1 ;
112 
113  if ( comp( *itB , *itA )) std::swap ( *itA, *itB);
114  if ( comp (*itC , *itB))
115  { std::swap (*itC , *itB );
116  if ( comp( *itB , *itA )) std::swap ( *itA, *itB);
117  };
118  std::swap ( *first , *itB);
119  value_t & val = const_cast < value_t &>(* first);
120  iter_t c_first = first+2 , c_last = last-2;
121 
122  while ( c_first != last and comp (*c_first, val)) ++c_first ;
123  while ( comp(val ,*c_last ) ) --c_last ;
124  while (not( c_first > c_last ))
125  { std::swap ( *(c_first++), *(c_last--));
126  while ( comp(*c_first,val) ) ++c_first;
127  while ( comp(val, *c_last) ) --c_last ;
128  }; // End while
129  std::swap ( *first , *c_last);
130 
131  //----------------------------------------------------------------
132  hpx::future <void> Rt1 = hpx::async ( &parallel_sort_comp::sort_thread,
133  this, first, c_last );
134 
135  hpx::future<void> Rt2 = hpx::async ( &parallel_sort_comp::sort_thread,
136  this,c_first, last );
137  return dataflow
138  ( [] (hpx::future<void> f1, hpx::future<void> f2) -> void
139  { f1.get();
140  f2.get();
141  return;
142  }, std::move(Rt1), std::move(Rt2) );
143  };
